### A. PLANETARY HORA CALCULATION
```python
def calculate_planetary_hora(date, location):
    # Sunrise calculation
    sunrise = calculate_sunrise(date, location)
    
    # Planetary sequence
    day_rulers = ['Sun', 'Moon', 'Mars', 'Mercury', 'Jupiter', 'Venus', 'Saturn']
    night_rulers = ['Jupiter', 'Venus', 'Saturn', 'Sun', 'Moon', 'Mars', 'Mercury']
    
    # Calculate current hora
    current_time = datetime.now()
    hours_since_sunrise = (current_time - sunrise).total_seconds() / 3600
    
    if hours_since_sunrise < 12:  # Day time
        hora_index = int(hours_since_sunrise) % 7
        return day_rulers[hora_index]
    else:  # Night time
        hora_index = int(hours_since_sunrise - 12) % 7
        return night_rulers[hora_index]
```

### B. BIORHYTHM CALCULATION FORMULAS

```python
import math
from datetime import datetime, date

def calculate_biorhythms(birth_date, target_date=None):
    if target_date is None:
        target_date = date.today()
    
    days_alive = (target_date - birth_date).days
    
    # Standard biorhythm cycles
    physical = math.sin(2 * math.pi * days_alive / 23)
    emotional = math.sin(2 * math.pi * days_alive / 28)
    intellectual = math.sin(2 * math.pi * days_alive / 33)
    
    # Additional cycles
    intuitive = math.sin(2 * math.pi * days_alive / 38)
    aesthetic = math.sin(2 * math.pi * days_alive / 43)
    spiritual = math.sin(2 * math.pi * days_alive / 53)
    
    return {
        'physical': physical,
        'emotional': emotional,
        'intellectual': intellectual,
        'intuitive': intuitive,
        'aesthetic': aesthetic,
        'spiritual': spiritual
    }
```
