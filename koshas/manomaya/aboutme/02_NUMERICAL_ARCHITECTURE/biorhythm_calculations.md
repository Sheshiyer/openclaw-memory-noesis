class BiorhythmCalculationEngine {
    constructor(birthDate, birthTime = '00:00') {
        this.birthDate = new Date(birthDate + 'T' + birthTime);
        this.cycles = {
            // Primary Cycles
            physical: 23,
            emotional: 28,
            intellectual: 33,
            
            // Secondary Cycles
            spiritual: 53,
            awareness: 38,
            aesthetic: 43,
            
            // Master Cycles
            intuitive: 38,
            passion: 53,
            
            // Consciousness Cycles
            transcendence: 62,
            unity: 89,
            
            // Cosmic Cycles
            solar: 365.25,
            lunar: 29.53
        };
    }
    
    calculateDaysLived(targetDate = new Date()) {
        const msPerDay = 24 * 60 * 60 * 1000;
        return Math.floor((targetDate - this.birthDate) / msPerDay);
    }
    
    calculateCycleValue(cycleName, targetDate = new Date()) {
        const daysLived = this.calculateDaysLived(targetDate);
        const cycleLength = this.cycles[cycleName];
        return Math.sin(2 * Math.PI * daysLived / cycleLength);
    }
    
    calculateAllCycles(targetDate = new Date()) {
        const results = {};
        for (const [cycleName, cycleLength] of Object.entries(this.cycles)) {
            const value = this.calculateCycleValue(cycleName, targetDate);
            const percentage = Math.round((value + 1) * 50); // Convert to 0-100%
            const phase = this.getCyclePhase(cycleName, targetDate);
            
            results[cycleName] = {
                value: value,
                percentage: percentage,
                phase: phase,
                cycleLength: cycleLength,
                daysInCycle: this.calculateDaysLived(targetDate) % cycleLength
            };
        }
        return results;
    }
    
    getCyclePhase(cycleName, targetDate = new Date()) {
        const daysLived = this.calculateDaysLived(targetDate);
        const cycleLength = this.cycles[cycleName];
        const dayInCycle = daysLived % cycleLength;
        const quarterCycle = cycleLength / 4;
        
        if (dayInCycle < quarterCycle) return 'Rising';
        if (dayInCycle < quarterCycle * 2) return 'Peak';
        if (dayInCycle < quarterCycle * 3) return 'Declining';
        return 'Low';
    }
    
    findCriticalDays(cycleName, daysAhead = 30) {
        const criticalDays = [];
        const cycleLength = this.cycles[cycleName];
        const today = new Date();
        
        for (let i = 0; i <= daysAhead; i++) {
            const checkDate = new Date(today.getTime() + i * 24 * 60 * 60 * 1000);
            const daysLived = this.calculateDaysLived(checkDate);
            const dayInCycle = daysLived % cycleLength;
            
            // Critical days are at 0, quarter, half, three-quarter points
            const criticalPoints = [0, cycleLength/4, cycleLength/2, (3*cycleLength)/4];
            
            for (const point of criticalPoints) {
                if (Math.abs(dayInCycle - point) < 0.5) {
                    criticalDays.push({
                        date: checkDate,
                        type: this.getCriticalType(point, cycleLength),
                        intensity: this.calculateCriticalIntensity(dayInCycle, point)
                    });
                }
            }
        }
        return criticalDays;
    }
    
    getCriticalType(point, cycleLength) {
        if (point === 0) return 'Cycle Start';
        if (point === cycleLength/4) return 'Rising Critical';
        if (point === cycleLength/2) return 'Peak Critical';
        if (point === (3*cycleLength)/4) return 'Declining Critical';
        return 'Unknown';
    }
    
    calculateCriticalIntensity(dayInCycle, criticalPoint) {
        const distance = Math.abs(dayInCycle - criticalPoint);
        return Math.max(0, 1 - distance); // Closer to critical point = higher intensity
    }
    
    generateOptimalTimingReport(daysAhead = 30) {
        const report = {
            highEnergyDays: [],
            lowEnergyDays: [],
            criticalDays: [],
            optimalActivities: {}
        };
        
        for (let i = 0; i <= daysAhead; i++) {
            const checkDate = new Date(Date.now() + i * 24 * 60 * 60 * 1000);
            const cycles = this.calculateAllCycles(checkDate);
            
            // Calculate overall energy
            const overallEnergy = (cycles.physical.value + cycles.emotional.value + cycles.intellectual.value) / 3;
            
            if (overallEnergy > 0.5) {
                report.highEnergyDays.push({
                    date: checkDate,
                    energy: overallEnergy,
                    dominantCycle: this.getDominantCycle(cycles)
                });
            } else if (overallEnergy < -0.5) {
                report.lowEnergyDays.push({
                    date: checkDate,
                    energy: overallEnergy,
                    recommendation: 'Rest and reflection'
                });
            }
            
            // Check for critical days across all cycles
            for (const cycleName of Object.keys(this.cycles)) {
                const criticals = this.findCriticalDays(cycleName, 1);
                if (criticals.length > 0) {
                    report.criticalDays.push(...criticals.map(c => ({...c, cycle: cycleName})));
                }
            }
        }
        
        return report;
    }
    
    getDominantCycle(cycles) {
        let maxValue = -2;
        let dominantCycle = '';
        
        for (const [cycleName, cycleData] of Object.entries(cycles)) {
            if (cycleData.value > maxValue) {
                maxValue = cycleData.value;
                dominantCycle = cycleName;
            }
        }
        
        return dominantCycle;
    }
}
