        return this.selectOptimalArchetype(archetypeWeights);

    calculateArchetypeWeights(biorhythm, numerology, humanDesign, tarot) {
        const weights = {};
        
        // Biorhythm influence on archetypes
        this.archetypes.primary.forEach(archetype => {
            let weight = 0;
            
            // Element alignment
            switch(archetype.element) {
                case 'Fire':
                    weight += biorhythm.physical * 0.3 + biorhythm.spiritual * 0.2;
                    break;
                case 'Water':
                    weight += biorhythm.emotional * 0.3 + biorhythm.intuitive * 0.2;
                    break;
                case 'Air':
                    weight += biorhythm.intellectual * 0.3 + biorhythm.mental * 0.2;
                    break;
                case 'Earth':
                    weight += (100 - Math.abs(biorhythm.physical)) * 0.005;
                    break;
            }
            
            // Planetary alignment
            if (humanDesign.dominantPlanet === archetype.planet) {
                weight += 25;
            }
            
            // Tarot correspondence
            if (tarot.name === archetype.tarotCard) {
                weight += 30;
            }
            
            // Numerological resonance
            if (numerology.personalYear % 12 === this.archetypes.primary.indexOf(archetype)) {
                weight += 15;
            }
            
            weights[archetype.name] = weight;
        });
        
        return weights;
    }

        const sortedArchetypes = Object.entries(weights)
            .sort(([,a], [,b]) => b - a)
            .map(([name]) => this.archetypes.primary.find(arch => arch.name === name));
        
        return sortedArchetypes[0];
    }

    getEvolutionPath(startArchetype, targetArchetype) {
        const path = this.evolutionPaths.find(path => 
            path.start === startArchetype && path.target === targetArchetype
        );
        
        if (!path) {

        return {
            phases: path.phases,
            duration: path.estimatedDuration,

        };
